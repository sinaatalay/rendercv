\newcommand{\resumeItem}[2]{
  \item\small{
    \ifthenelse{\equal{#1}{}}{#2}{\textbf{#1:}{ #2}}
  }
}

\newcommand{\resumeItemWithoutBullet}[2]{
  \item[]\small{
    \ifthenelse{\equal{#1}{}}{#2}{\textbf{#1:}{ #2}}
  }
}

\newcommand{\resumeSubheading}[4]{

      #1 & \small\ifthenelse{\equal{#2}{}}{#4}{#2} \\
      {\small#3} & \small\ifthenelse{\equal{#2}{}}{}{#4} \\
    \end{tabularx}
}
